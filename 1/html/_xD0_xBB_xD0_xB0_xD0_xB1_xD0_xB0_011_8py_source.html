21def sift_down(arr,i,last_index):
22 while True:
23 left_j = 2*i + 1 # левый ребёнок
24 right_j = 2*i + 2 # правый ребёнок
25 j=i # родитель
26 if left_j < last_index and arr[left_j] > arr[j]:
27 j = left_j
28 if right_j < last_index and arr[right_j] > arr[j]:
29 j = right_j
30 if j != i: # если какой-то из ребёнков больше чем родитель

37def heap_sort(mas):
38 arr=mas.copy()
39 sift_up(arr)
40 last_index = len(arr) - 1
41 while last_index > 0:
42 arr[0], arr[last_index] = arr[last_index], arr[0]
43 sift_down(arr, 0, last_index)
44 last_index -=1
45 return arr
